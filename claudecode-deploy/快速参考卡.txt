1️⃣ 拷贝文件
   cp -r /mnt/usb/claudecode-deploy ~/
   cd ~/claudecode-deploy

2️⃣ 移动 zip
   sudo mv claude-docs.zip /tmp/

【常用命令】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

# 查看服务状态
sudo systemctl status nginx

# 查看错误日志
sudo tail -f /var/log/nginx/claude-docs-error.log

# 查看部署文件
ls -la /var/www/claude-docs/

# 重启 Nginx
sudo systemctl restart nginx

# 回滚到备份
sudo rm -rf /var/www/claude-docs
sudo cp -r /var/www/claude-docs.backup.YYYYMMDD /var/www/claude-docs
sudo nginx -s reload

【故障速查】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

问题：无法访问网站
解决：
  1. ping <服务器IP>
  2. sudo systemctl status nginx
  3. 检查 hosts 文件

问题：403 Forbidden
解决：
  sudo chown -R nginx:nginx /var/www/claude-docs
  sudo chmod -R 755 /var/www/claude-docs

问题：更新后看不到变化
解决：
  浏览器按 Ctrl + F5 强制刷新
